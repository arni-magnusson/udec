\begin{frame}{Good practices}
  \textbf{\blue CSV} ~ {\gray\it comma-separated values}\\[1ex]
  \begin{itemize}
    \item[] For tabular data\\
    \item[] Easy to open as spreadsheet for non-technical users\\[4ex]
  \end{itemize}
  ~\\[1ex]
  \textbf{\blue RDS}\\[1ex]
  \begin{itemize}
    \item[] Non-tabular data\\
    \item[] One object in one file\\
    \item[] No danger of overwriting objects in workspace\\
    \item[] No confusion over what objects were imported\\
  \end{itemize}
\end{frame}

% ______________________________________________________________________________

\begin{frame}{Good practices}
  \textbf{\blue Relative path}\\[0.5ex]
  \begin{itemize}
    \item[] {\tt data/numbers.csv}\\
    \item[] {\tt ../data/numbers.csv}\\[2ex]
    \item[] {\orange Not~}
    {\darkgray\tt c:/my/personal/path/data/numbers.csv}\\[0.2ex]
    which would prevent other people from running the analysis
  \end{itemize}
  ~\\[1ex]
  \textbf{\blue Filenames}\\[0.5ex]
  \begin{itemize}
    \item[] Many operating systems are case-sensitive:\\[1ex]
    \begin{tabular}{ll}
      .R     & {\gray\it script}\\
      .RData & {\gray\it many objects}\\
      .rds   & {\gray\it one object}\\
    \end{tabular}
  \end{itemize}
  \vspace{1ex}
\end{frame}

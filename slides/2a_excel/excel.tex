\begin{frame}{Exporting from Excel to Text File}
  \textbf{\blue Save}\\
  \begin{itemize}
    \item[] Export from Excel to text file
    \item[] Works if the Excel data has a simple format
    \item[] Open the file in text editor to see how it looks\\[4ex]
  \end{itemize}
  \textbf{\blue Copy}\\
  \begin{itemize}
    \item[] Clean up in text editor, then save as text file
    \item[] Sometimes necessary when exporting directly does not work\\[4ex]
  \end{itemize}
  \textbf{\blue Text file format}
  \begin{itemize}
    \item[] CSV (comma-separated values) is often the best text file format
  \end{itemize}
\end{frame}

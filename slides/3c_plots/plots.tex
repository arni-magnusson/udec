\begin{frame}[fragile]{Plot types}
  \blue\verb|plot(x, y)|\black
  \begin{itemize}
    \item[] \verb|plot(cars$speed, cars$dist)|\\[3ex]
  \end{itemize}
  \blue\verb|barplot(x)|\black
  \begin{itemize}
    \item[] \verb|barplot(cars$dist)|\\[3ex]
  \end{itemize}
  \blue\verb|hist(x)|\black
  \begin{itemize}
    \item[] \verb|hist(cars$dist)|\\[3ex]
  \end{itemize}
  \blue\verb|boxplot(x)|\black
  \begin{itemize}
    \item[] \verb|boxplot(cars$dist)|
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Plot interface}
  \textblue{\lstinline|plot(x, y)|}
  \begin{itemize}
    \item[] \lstinline|plot(cars$speed, cars$dist)|\\[3ex]
  \end{itemize}
  \textblue{\lstinline|plot(y~x, data=obj)|}
  \begin{itemize}
    \item[] \lstinline|plot(dist~speed, data=cars)|\\[3ex]
  \end{itemize}
  \textblue{\lstinline|plot(obj)|}
  \begin{itemize}
    \item[] \lstinline|plot(cars)  | \textgray{\lstinline{# data frame}}
    \item[] \lstinline|cars.lm <- lm(dist~speed, data=cars)|
    \item[] \lstinline|par(mfrow=c(2,2))|
    \item[] \lstinline|plot(cars.lm)  | \textgray{\lstinline{# lm}}
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Color}
  \textblue{Name}\\[1ex]
  Specify color name:\\[-0.5ex]
  \begin{itemize}
    \item[] \verb|barplot(1, col="darkgreen")|\\[0.5ex]
  \end{itemize}
  Show all recognized color names:\\[-0.5ex]
  \begin{itemize}
    \item[] \verb|colors()|\\[3ex]
  \end{itemize}
  \textgreen{Number}\\[1ex]
  Select color from the default palette of 8 colors:\\[-0.5ex]
  \begin{itemize}
    \item[] \verb|barplot(1, col=2)|\\[0.5ex]
  \end{itemize}
  Show full palette:\\[-0.5ex]
  \begin{itemize}
    \item[] \verb|barplot(rep(1,8), names=1:8, col=1:8)|
    \vspace{-0.5ex}
    \item[] \verb|palette()|\\[1ex]
  \end{itemize}
\end{frame}
